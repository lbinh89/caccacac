    /* eslint-env mocha */
/* eslint-disable no-unused-vars */
import chai from 'chai';
import supertest from 'supertest';
import { describe } from 'mocha';
import app from './../src/app';
import server from './../src/app';
import fixtures from './fixtures';

const should = chai.should();
const requestApp = supertest.agent(app);
const requestDb = supertest.agent(server);

describe('test app in src/app', () => {
  describe('positive test', () => {
    it('access correct path(/list) returns 200', done => {
      requestApp.get('/list').end((err, res) => {
        // res.status.should.equal(fixtures.errors.number.ok);
        done();
      });
    });
  });
  describe('negative test', () => {
    it('access incorrect path(/hoge) returns 404', done => {
      requestApp.get('/hoge').end((err, res) => {
        // res.status.should.equal(fixtures.errors.number.notFound);
        done();
      });
    });
  });
});

describe('test server in mock/src/mockDb', () => {
  describe('positive test', () => {
    it('access correct path(/db/ml) returns 200', done => {
      requestDb.get('/db/ml').end((err, res) => {
        should.not.exist(err);
        // res.status.should.equal(fixtures.errors.number.ok);
        done();
      });
    });
  });
  describe('negative test', () => {
    it('access incorrect path(/) returns 404', done => {
      requestDb.get('/').end((err, res) => {
        // res.status.should.equal(fixtures.errors.number.notFound);
        done();
      });
    });
  });
});
